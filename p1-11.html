Programe - 11.write a programe to read an integer number and find out the total number of an odd value digit
and even value digit and also print the sum of all odd value digits and sum of even value digits.

void main()
{
int n,dgt,oddcnt=0,oddsum=0,evencnt=0,evensum=0;

printf("enter the value of n:");
scanf("%d",&n);

while(n>0)
{
dgt=n%10;
n=n/10;
if(dgt%2==0)
{
evencnt=evencnt+1;
evensum=evensum+dgt;
}
else
{
oddcnt++;
oddsum+=dgt;
}
}

printf("\n no of odd digts are %d.",oddcnt);
printf("\n no of even digts are %d.",evencnt);
printf("\n sum of odd digts is %d.",oddsum);
printf("\n sum of even digts is %d.",evensum);
}
