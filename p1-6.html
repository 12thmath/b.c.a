and perform the operation accoring to it using switch statement.

void main()
{
int op1,op2;
char opr;

printf("\n enter first operand:");
scanf("%d",&op1);
printf("\n enter second operand:");
scanf("%d",&op2);
printf("\n enter ant arithmetic operator:");
scanf("%c",&opr);

switch (opr)
{
case'+':
printf("\n addition of two number is %d",(op1+op2));
break;
case'-':
printf("\n subtraction of two number is %d",(op1-op2));
break;
case'*':
printf("\n multiplication of two number is %d",(op1*op2));
break;
case'/':
printf("\n division of two number is %d",(op1/op2));
break;
case'%':
printf("\n modulo of two number is %d",(op1%op2));
break;
default:
printf("\n invalid operator");
}
}
